8. Create new PV,new VG with 16MB extend, 50 extend and must vfat filesystem, then mount /data1
================================================================================================
# fdisk /dev/vda
> select "p"
> select "n"
> select "l"
> +1000M
> select t; 83 (linux LVM)
> w
# partprobe
# pvcreate /dev/vdaX
# vgcreate -s 16M vgdata /dev/vdaX
# vgdisplay vgdata  < -- refer at PE to verify
# lvdisplay -l 50 -n data1 vgdata
# mkfs.ext4 /dev/mapper/vgdata-data1
# cd / | mkdir /data1
# blkid
# vi /etc/fstab
UUID=xxxxxxx /data1 ext4 defaults 0 0
# mount -a
# df -hT  < -- to verify

10. Create 100MB PV under /data mountpoint
=========================================
*I remember this question they only have 1 disk with 3 PV(/dev/sda1, /dev/sda2, /dev/sda3). Read carefully next question you need configure new PV for swap. Please remember to select /dev/sda then select "extend" for partition number 4 to use for VG and next wuestion for swap. Below is sample:

# fdisk /dev/vdb
> select "n"
> choose extend
> enter
> +20GB  < -- set to max size for extend
> w
# partprobe  < -- to save partition without reboot
# fdisk /dev/vdb
> select "n"
> select "l"
> enter
> +100MB
> w
# partprobe
# mkfs.xfs /dev/vdbX
# blkid  < -- check ID device after format
# cd / | mkdir /data
# vi /etc/fstab
UUID=xxxxxxx /data xfs defaults 0 0
# mount -a

11. Create new 100MB for SWAP
=============================
# free -m
# fdisk /dev/vdb
> select "p"
> select "n"
> select "l"
> +100M
> select t; 82 (linux swap)  < -- please remember set swap type for this partition
> p  < --- to is partition table to view for verify
> w
# partprobe
# free -m
# mkswap /dev/vdaX
# swapon /dev/vdaX
# swapon -a
# swapon -s
# swapon -p 1 /dev/vdaX
# vi /etc/fstab
> uuid	swap swap pri=1 0 0    < --- pri is set priority for swap device and persistent after reboot
# free -m
